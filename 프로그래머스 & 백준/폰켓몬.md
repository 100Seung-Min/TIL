### 폰켓몬
```java
class Solution {
    public int solution(int[] nums) {
        int answer = 0;
        int[] array = new int[nums.length];
        for(int i = 0; i < nums.length; i++){
            if(!contains(array, nums[i]) && nums.length / 2 > answer){
                array[answer] = nums[i];
                answer++;
            }
        }
        return answer;
    }

    boolean contains(int[] arr, int num){
        for(int a: arr){
            if(a == num){
                return true;
            }
        }
        return false;
    }
}
```
