### 로또 최고 순위와 최저 순위
```java
class Solution {
    fun solution(lottos: IntArray, win_nums: IntArray): IntArray {
        var answer: IntArray = intArrayOf()
        var zeroCnt = 0
        var cnt = 0
        for(i in 0..lottos.size - 1){
            if(lottos[i] == 0){
                zeroCnt++
            } else if(win_nums.find{it == lottos[i]} != null){
                cnt++
            }
        }
        answer = answer.plus(if(7 - zeroCnt-cnt == 7) 6 else 7 - zeroCnt-cnt)
        answer = answer.plus(if(7 - cnt == 7) 6 else 7 - cnt)
        return answer
    }
}
```
