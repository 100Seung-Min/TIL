### 카드 뭉치
```java
class Solution {
    fun solution(cards1: Array<String>, cards2: Array<String>, goal: Array<String>): String {
        var cards1Index = 0
        var cards2Index = 0
        var goalIndex = 0
        var wrongCount = 0
        if(cards1.size + cards2.size < goal.size) return "No"
        while(goalIndex <= goal.size - 1) {
            if(cards1[cards1Index] == goal[goalIndex]) {
                if(cards1Index != cards1.size - 1) cards1Index++
            } else if(cards2[cards2Index] == goal[goalIndex]) {
                if(cards2Index != cards2.size - 1) cards2Index++
            } else {
                cards1Index = 0
                cards2Index = 0
                goalIndex = 0
                wrongCount++
                break
            }
            goalIndex++
        }
        while(goalIndex <= goal.size - 1) {
            if(cards2[cards2Index] == goal[goalIndex]) {
                if(cards2Index != cards2.size - 1) cards2Index++
            } else if(cards1[cards1Index] == goal[goalIndex]) {
                if(cards1Index != cards1.size - 1) cards1Index++
            } else {
                wrongCount++
                break
            }
            goalIndex++
        }
        return if (wrongCount == 2) "No" else "Yes"
    }
}
```
