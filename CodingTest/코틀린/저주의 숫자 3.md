### 저주의 숫자 3
```java
class Solution {
    fun solution(n: Int): Int {
        var answer = 1
        for (i in 1 until n) {
            answer += 1
            while (true) {
                if (answer % 3 == 0 || answer.toString().contains("3")) {
                    answer++
                } else {
                    break
                }
            }
        }
        return answer
    }
}
```
