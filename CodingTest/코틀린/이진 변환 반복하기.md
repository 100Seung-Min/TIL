### 이진 변환 반복하기
```java
class Solution {
    fun solution(s: String): IntArray = check(s, intArrayOf(0, 0))

    tailrec fun check(s: String, arr: IntArray): IntArray = if(s == "1") arr else check(s.filter { it != '0' }.length.toString(2), intArrayOf(arr[0] + 1, arr[1] + s.count { it == '0' }))
}
```
