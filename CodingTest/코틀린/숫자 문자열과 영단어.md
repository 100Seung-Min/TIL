### 숫자 문자열과 영단어
```java
class Solution {
    fun solution(s: String): Int {
        var answer = s
        arrayOf("zero", "one", "two", "three", "four", "five", "six", "seven", "eight", "nine").mapIndexed { i, v ->
            answer = answer.replace(
                v,
                i.toString()
            )
        }
        return answer.toInt()
    }
}
```
