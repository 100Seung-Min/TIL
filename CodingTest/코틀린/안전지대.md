### 안전지대
```java
class Solution {
    fun solution(board: Array<IntArray>): Int {
        val answerBoard = board.map{it.copyOf()}.toTypedArray()
        board.forEachIndexed { i, raw ->
            raw.forEachIndexed {j, box -> 
                if(box == 1) {
                    if(i != 0) answerBoard[i - 1][j] = 1
                    if(i != 0 && j != board[i].size - 1) answerBoard[i - 1][j + 1] = 1
                    if(i != 0 && j != 0) answerBoard[i - 1][j - 1] = 1
                    if(j != board[i].size - 1) answerBoard[i][j + 1] = 1
                    if(j != 0) answerBoard[i][j - 1] = 1
                    if(i != board.size - 1)answerBoard[i + 1][j] = 1
                    if(i != board.size - 1 && j != board[i].size - 1)answerBoard[i + 1][j + 1] = 1
                    if(i != board.size - 1 && j != 0)answerBoard[i + 1][j - 1] = 1
                }
            }
        }
        return answerBoard.sumOf{it.count{it==0}}
    }
}
```
