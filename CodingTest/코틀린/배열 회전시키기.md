### 배열 회전시키기
```java
class Solution {
    fun solution(numbers: IntArray, direction: String): IntArray = when(direction) {
        "right" -> {
            numbers.mapIndexed { i, _ ->
                if (i == 0) {
                    numbers[numbers.lastIndex]
                } else {
                    numbers[i - 1]
                }
            }.toIntArray()
        }
        else -> {
            numbers.mapIndexed { i, _ ->
                if (i == numbers.lastIndex) {
                    numbers[0]
                } else {
                    numbers[i + 1]
                }
            }.toIntArray()
        }
    }
}
```
