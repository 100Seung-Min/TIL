### 정수를 나선형으로 배치하기
```java
class Solution {
    fun solution(n: Int): Array<IntArray> {
        var answer = Array(n){IntArray(n)}
        var number = 1
        var i = 0
        var j = 0
        var direction = 0
        var top = 1
        var bottom = n - 1
        var start = 0
        var end = n - 1
        while(true) {
            answer[i][j] = number
            if(number == n*n) break
            number++
            when(direction) {
                0 -> if(j == end) {
                    direction = 1
                    end -= 1
                    i++
                } else j++
                1 -> if(i == bottom) {
                    direction = 2
                    bottom -= 1
                    j--
                } else i++
                2 -> if(j == start) {
                    direction = 3
                    start += 1
                    i--
                } else j--
                3 -> if(i == top) {
                    direction = 0
                    top += 1
                    j++
                } else i--
            }
        }
        return answer
    }
}
```
