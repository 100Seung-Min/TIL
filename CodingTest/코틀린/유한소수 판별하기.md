### 유한소수 판별하기
```java
class Solution {
    fun solution(a: Int, b: Int): Int = divide(a, b, 2).let { if (it[1].check() || it[1] == 1) 1 else 2 }

    fun Int.check(): Boolean {
        var number = this
        var answer = listOf<Int>()
        var compare = listOf<Int>()
        (2..this).forEach {
            while (number % it == 0) {
                if (it == 2 || it == 5) {
                    answer = answer.plus(it)
                }
                compare = compare.plus(it)
                number /= it
            }
        }
        return answer.size == compare.size
    }

    tailrec fun divide(n1: Int, n2: Int, a: Int): IntArray = if (a > n1 || a > n2) intArrayOf(n1, n2) else if (n1 % a == 0 && n2 % a == 0) divide(n1 / a, n2 / a, a) else divide(n1, n2, a + 1)
}
```
