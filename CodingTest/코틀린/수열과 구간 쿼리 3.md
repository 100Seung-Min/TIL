### 수열과 구간 쿼리 3
```java
class Solution {
    fun solution(arr: IntArray, queries: Array<IntArray>): IntArray {
        var answer: IntArray = arr
        queries.forEach {
            var temp = answer[it[0]]
            answer[it[0]] = answer[it[1]]
            answer[it[1]] = temp
        }
        return answer
    }
}
```
