### 예상 대진표
```java
class Solution {
    fun solution(n: Int, a: Int, b: Int): Int = if(a.isBattle(b)) 1 else check(if(a % 2 == 0) a/2 else (a+1)/2, if(b % 2 == 0) b/2 else (b+1)/2, 2)
    
    tailrec fun check(a: Int, b: Int, cnt: Int): Int = if(a.isBattle(b)) cnt else check(if(a % 2 == 0) a/2 else (a+1)/2, if(b % 2 == 0) b/2 else (b+1)/2, cnt+1)
    
    fun Int.isBattle(rival: Int): Boolean = (this - rival).abs() == 1 && arrayOf(this, rival).sorted().last() % 2 == 0
    
    fun Int.abs() = if(this < 0) this * -1 else this
}
```
