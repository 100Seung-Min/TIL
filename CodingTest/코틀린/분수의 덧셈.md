### 분수의 덧셈
```java
class Solution {
    fun solution(numer1: Int, denom1: Int, numer2: Int, denom2: Int): IntArray = divide(numer1 * denom2 + numer2 * denom1, denom1 * denom2, 2)

    tailrec fun divide(n1: Int, n2: Int, a: Int): IntArray = if (a > n1 || a > n2) intArrayOf(n1, n2) else if (n1 % a == 0 && n2 % a == 0) divide(n1 / a, n2 / a, a) else divide(n1, n2, a + 1)
}
```
