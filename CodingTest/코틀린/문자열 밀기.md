### 문자열 밀기
```java
class Solution {
    fun solution(A: String, B: String): Int {
        if (A == B) return 0
        for (i in 1..A.lastIndex) {
            if ("${A.slice(i..A.lastIndex)}${A.slice(0 until i)}" == B) {
                return A.length - i
                break
            }
        }
        return -1
    }
}
```
