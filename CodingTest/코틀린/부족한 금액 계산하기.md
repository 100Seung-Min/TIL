### 부족한 금액 계산하기
```java
class Solution {
    fun solution(price: Int, money: Int, count: Int): Long {
        var realPrice:Long = 0L
        for(i in 1..count){
            realPrice += price * i
        }
        return if(money - realPrice > 0) 0L else realPrice - money
    }
}
```
