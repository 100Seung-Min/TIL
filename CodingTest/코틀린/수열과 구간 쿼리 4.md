### 수열과 구간 쿼리 4
```java
class Solution {
    fun solution(arr: IntArray, queries: Array<IntArray>): IntArray {
        var answer = arr
        queries.forEach {
            (it[0]..it[1]).forEach { v ->
                if(v % it[2] == 0) {
                    answer[v]++
                }
            }
        }
        return answer
    }
}
```
