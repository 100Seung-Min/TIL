### 두 개 뽑아서 더하기
```java
class Solution {
    fun solution(numbers: IntArray): IntArray {
        var answer = listOf<Int>()
        for (i in 0 until numbers.lastIndex)
            for (j in i + 1..numbers.lastIndex)
                answer = answer.plus(numbers[i] + numbers[j])
        return answer.toSortedSet().toIntArray()
    }
}
```
