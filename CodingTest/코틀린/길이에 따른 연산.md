### 길이에 따른 연산
```java
class Solution {
    fun solution(num_list: IntArray): Int = if (num_list.size <= 10) num_list.multiple() else num_list.sum()
    
    fun IntArray.multiple(): Int {
        var result = 1
        this.forEach {
            result *= it
        }
        return result
    }
}
```
