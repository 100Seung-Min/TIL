### 주차 요금 계산
```java
class Solution {
    fun solution(fees: IntArray, records: Array<String>): IntArray {
        var answer = listOf<Int>()
        records.map { it.split(" ") }.sortedBy { it[1].toInt() }
            .let {
                var totalTime = 0
                it.forEachIndexed { i, v ->
                    if (v[2] == "OUT") totalTime += v[0].toTime()
                    else if (v[2] == "IN") totalTime -= v[0].toTime()
                    if (i == it.lastIndex || v[1] != it[i + 1][1]) {
                        if ( totalTime <= 0) totalTime += 1439
                        var money = fees[1] + if (totalTime - fees[0] > 0) ceil((totalTime - fees[0]) / fees[2].toFloat()).toInt() * fees[3] else 0
                        answer = answer.plus(money)
                        totalTime = 0
                    }
                }
            }
        return answer.toIntArray()
    }

    fun String.toTime() = (this.slice(0..1).toInt() * 60) + this.slice(3..4).toInt()
}
```
