### 조건 문자열
```java
class Solution {
    fun solution(ineq: String, eq: String, n: Int, m: Int): Int = when(ineq) {
        ">" -> {
            if(eq == "=" && n >= m) 1
            else if(eq == "!" && n > m) 1
            else 0
        }
        else -> {
            if(eq == "=" && n <= m) 1
            else if(eq == "!" && n < m) 1
            else 0
        }
    }
}
```
