### 배열의 길이를 2의 거듭제곱으로 만들기
```java
class Solution {
    fun solution(arr: IntArray): IntArray {
        var array = arr
        if(arr.size == 1) return arr
        (1..twoPow(2, arr.size)).forEach {array = array.plus(0)}
        return array
    }
    
    tailrec fun twoPow(n: Int, size: Int): Int = if(n >= size) n - size else twoPow(n * 2, size)
}
```
