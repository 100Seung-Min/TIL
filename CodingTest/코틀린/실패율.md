### 실패율
```java
class Solution {
    fun solution(N: Int, stages: IntArray): IntArray = (1..N)
            .map{ stage -> if(stages.count{it >= stage} == 0) arrayOf(0F, stage.toFloat()) else arrayOf(stages.count{it == stage}.toFloat() / stages.count{it >= stage}.toFloat(), stage.toFloat()) }
            .sortedByDescending{it[0]}
            .map{it[1].toInt()}
            .toIntArray()
}
```
