### 다항식 더하기
```java
class Solution {
    fun solution(polynomial: String): String {
        val list = polynomial.split(" + ")
        val xCount = list.filter { it.contains("x") }.sumOf { if(it == "x") 1 else it.replace("x", "").toInt() }
        val sum = list.filter { !it.contains("x") }.sumOf { it.toInt() }
        return when(xCount) {
            0 -> if (sum == 0) "0" else "$sum"
            1 -> if (sum == 0) "x" else "x + $sum"
            else -> if (sum == 0) "${xCount}x" else "${xCount}x + $sum"
        }
    }
}
```
