### 최빈값 구하기
```java
class Solution {
    fun solution(array: IntArray): Int {
        var max = 0
        var index = -1
        array.toSet().forEach { value ->
            array.count { value == it }.let {
                if (max < it) {
                    max = it
                    index = value
                }
                else if (max == it) index = -1
            }
        }
        return if (index == -1) -1 else array[array.indexOf(index)]
    }
}
```
