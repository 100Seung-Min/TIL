class Solution {
    fun solution(babbling: Array<String>): Int {
        var answer: Int = 0
        var wordList = arrayOf("aya", "ye", "woo", "ma")
        babbling.forEach {
            var replaceWord = it
            wordList.forEach {word ->
                replaceWord = replaceWord.replaceFirst(word, " ")
            }
            if (replaceWord.trim().isEmpty()) {
                answer++
            }
        }
        return answer
    }
}
