class Solution {
    fun solution(a: Int, b: Int, c: Int, d: Int): Int = arrayOf(a, b, c, d).let{
        when(it.toSet().size) {
            1 -> it[0] * 1111
            2 -> {
                if(it.toSet().sum() * 2 == it.sum()) (it.toSet().first() + it.toSet().last()) * (it.toSet().first() - it.toSet().last()).abs()
                else (10 * it.toSet().filter { v -> it.count { it == v } == 3 }.first() + it.toSet().filter { v -> it.count { it == v } != 3 }.first()).pow()
            }
            3 -> it.toSet().filter { v -> it.count { it == v } != 2 }.first() * it.toSet().filter { v -> it.count { it == v } != 2 }.last()
            else -> it.minOrNull() ?: 0
        }
    }
    
    fun Int.pow() = this * this
    fun Int.abs() = if(this < 0) this * -1 else this
}
