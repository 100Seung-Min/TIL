### 정사각형으로 만들기
```java
class Solution {
    fun solution(arr: Array<IntArray>): Array<IntArray> {
        var answer: Array<IntArray> = arrayOf()
        var diff = abs(arr.size - arr[0].size)
        if(diff == 0) return arr
        if(arr.size > arr[0].size) {
            for(i in 0 until arr.size) {
                var addRow = arr[i]
                (0 until diff).forEach{addRow = addRow.plus(0)}
                answer = answer.plus(addRow)
            }
        } else {
            answer = arr
            for(i in 0 until diff) {
                answer = answer.plus(IntArray(arr[0].size){0})
            }
        }
        return answer
    }
}
```
