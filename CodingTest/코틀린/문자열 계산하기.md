### 문자열 계산하기
```java
class Solution {
    fun solution(my_string: String): Int {
        var answer = 0
        var arr = my_string.split(" ")
        var operator = "+"
        arr.forEachIndexed { i, s ->
            if (i % 2 == 0) {
                if (operator == "+")
                    answer += s.toInt()
                else
                    answer -= s.toInt()
            } else {
                operator = s
            }
        }
        return answer
    }
}
```
