### 배열 조각하기
```java
class Solution {
    fun solution(arr: IntArray, query: IntArray): IntArray {
        var answer = arr
        query.forEachIndexed{ i, v ->
            answer = if(i % 2 == 0) answer.slice(0..v).toIntArray() else answer.slice(v..answer.lastIndex).toIntArray()
        }
        return answer
    }
}
```
