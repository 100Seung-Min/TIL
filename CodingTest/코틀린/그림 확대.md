### 그림 확대
```java
class Solution {
    fun solution(picture: Array<String>, k: Int): Array<String> {
        var answer = arrayOf<String>()
        picture.forEach {
            var addString = it.map{it.toString().repeat(k)}.joinToString("")
            (1..k).forEach{
                answer = answer.plus(addString)
            }
        }
        return answer
    }
}
```
