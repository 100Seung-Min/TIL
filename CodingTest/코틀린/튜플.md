class Solution {
    fun solution(s: String): IntArray {
        var answer = arrayOf<IntArray>()
        var numberList = s.filter { !arrayOf('{', '}').contains(it) }.split(",").map { it.toInt() }.toList()
        var numberCount = numberList.toSet().map { numberList.count { v -> v == it } }
        numberList.toSet().forEachIndexed { i, v ->
            answer = answer.plus(intArrayOf(v, numberCount[i]))
        }
        return answer.sortedByDescending { it[1] }.map { it[0] }.toIntArray()
    }
}
