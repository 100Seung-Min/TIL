### 문자 개수 세기
```java
class Solution {
    fun solution(my_string: String): IntArray {
        var answer = Array<Int>(52) { 0 }
        my_string.forEach {
            if (it.isUpperCase()) answer[it.toInt() - 65] += 1
            else answer[it.toInt() - 71] += 1
        }
        return answer.toIntArray()
    }
}
```
