### 원소들의 곱과 합
```java
class Solution {
    fun solution(num_list: IntArray): Int = if (num_list.multiple() > num_list.sum().pow()) 0 else 1
    
    fun IntArray.multiple(): Int {
        var result = 1
        this.forEach {
            result *= it
        }
        return result
    }
    
    fun Int.pow(): Int = this * this
}
```
