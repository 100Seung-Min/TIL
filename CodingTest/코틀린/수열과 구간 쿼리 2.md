### 수열과 구간 쿼리 2
```java
class Solution {
    fun solution(arr: IntArray, queries: Array<IntArray>): IntArray {
        var answer: IntArray = intArrayOf()
        queries.forEach {
            answer = answer.plus(arr.slice(it[0]..it[1]).sorted().firstOrNull{v -> v > it[2]} ?: -1)
        }
        return answer
    }
}
```
