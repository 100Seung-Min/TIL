### 소인수분해
```java
class Solution {
    fun solution(n: Int): IntArray {
        var number = n
        var answer = listOf<Int>()
        (2..n).forEach {
            while (number % it == 0) {
                answer = answer.plus(it)
                number /= it
            }
        }
        return answer.toSet().toIntArray()
    }
}
```
