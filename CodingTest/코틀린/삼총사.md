### 삼총사
```java
class Solution {
    fun solution(number: IntArray): Int {
        var answer = 0
        for (i in 0..number.lastIndex - 2)
            for (j in i + 1 until number.lastIndex)
                for (h in j + 1..number.lastIndex)
                    if (number[i] + number[j] + number[h] == 0) answer++
        return answer
    }
}
```
