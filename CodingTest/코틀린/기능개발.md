### 기능개발
```java
class Solution {
    fun solution(progresses: IntArray, speeds: IntArray): IntArray {
        var answer = intArrayOf()
        var restProgresses = progresses.mapIndexed { i, v -> 
            Math.ceil((100 - v) / speeds[i].toDouble()).toInt()
        }
        var progressCnt = 0
        var lastProgress = restProgresses[0]
        restProgresses.forEach {
            if(lastProgress < it) {
                answer = answer.plus(progressCnt)
                progressCnt = 0
                lastProgress = it
            }
            progressCnt++
        }
        return answer.plus(progressCnt)
    }
}
```
