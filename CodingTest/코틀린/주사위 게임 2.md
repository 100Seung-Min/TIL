### 주사위 게임 2
```java
class Solution {
    fun solution(a: Int, b: Int, c: Int): Int = if(a != b && b != c && a != c) a + b + c else if(a == b && b == c) (a + b + c) * (pow(a, 2) + pow(b, 2) + pow(c, 2)) * (pow(a, 3) + pow(b, 3) + pow(c, 3)) else (a + b + c) * (pow(a, 2) + pow(b, 2) + pow(c, 2))
    
    fun pow(a: Int, b: Int): Int = a.toFloat().pow(b).toInt()
}
```
