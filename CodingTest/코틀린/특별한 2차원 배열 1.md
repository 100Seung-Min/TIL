### 특별한 2차원 배열 1
```java
class Solution {
    fun solution(n: Int): Array<IntArray> {
        var answer: Array<IntArray> = arrayOf<IntArray>()
        for (i in 0 until n) {
            answer = answer.plus(Array(n) {v -> if(i == v) 1 else 0}.toIntArray())
        }
        return answer
    }
}
```
