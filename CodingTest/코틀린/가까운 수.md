### 가까운 수
```java
class Solution {
    fun solution(array: IntArray, n: Int): Int {
        var answer: Int = array[0]
        array.forEach {
            if(abs(it - n)  <= abs(answer - n)) {
                answer = if (abs(it - n)  == abs(answer - n)) min(it, answer) else it
            }
        }
        return answer
    }
}
```
