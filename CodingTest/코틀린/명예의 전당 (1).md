class Solution {
    fun solution(k: Int, score: IntArray): IntArray {
        var answer = intArrayOf()
        var record = listOf<Int>()
        score.forEach {
            if(record.size < k || it > answer[answer.lastIndex]) {
                record = record.plus(it).sortedDescending()
                if (record.size >= k) record = record.slice(0 until k)
            }
            answer = answer.plus(record.last())
        }
        return answer
    }
}
